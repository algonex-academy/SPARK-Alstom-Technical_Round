11 . Write a code for factorial & even/odd of a number within a single function

DELIMITER //
CREATE FUNCTION factorial_even_odd(n INT)
RETURNS VARCHAR(255)
BEGIN
     result VARCHAR(255);
    IF n % 2 = 0 THEN
        SET result = CONCAT('Factorial: ', fact, ' | Even Number');
    ELSE
        SET result = CONCAT('Factorial: ', fact, ' | Odd Number');
    END IF;
    RETURN result;
END //
DELIMITER ;
CALL factorial_even_odd(n INT);
