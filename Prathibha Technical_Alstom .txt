10.Write a code for factorial & even/odd of a number within a single function
def number_operations(n):
    if n % 2 == 0:
        print(f"{n} is Even")
    else:
        print(f"{n} is Odd")
    if n < 0:
        print("Factorial is not defined for negative numbers")
    else:
        fact = 1
        for i in range(1, n + 1):
            fact *= i
        print(f"Factorial of {n} is {fact}")
