create table tickets(
ticket_id int auto_increment primary key,
title varchar(100),
status VARCHAR(20) DEFAULT 'Open'
);

  dept_id INT AUTO_INCREMENT PRIMARY KEY,
  dept_name VARCHAR(100) NOT NULL,

--  inserting values
INSERT INTO departments VALUES
(1,'IT'),
(2,'HR'),
(3,'Support');

INSERT INTO users VALUES
(101,1,'asha@gmail.com'),
(102,2,'ravi@gmail.com'),
(103,3,'john@gmail.com');

INSERT INTO tickets VALUES
(1001,'Login issue','Open'),
(1002,'Salary slip request','Closed'),
(1003,'System not working','Open');

-- updating values
UPDATE tickets
SET status = 'Closed'
WHERE ticket_id = 1001;

-- deleting the columns based on conditions
DELETE FROM tickets
WHERE ticket_id = 1003;

-- fetching data using join
SELECT t.ticket_id,
       t.title,
       u.user_name,
       d.dept_name
FROM tickets t
JOIN users u 
ON t.ticket_id = u.ticket_id
JOIN departments d 
ON t.dept_id = d.dept_id;

-- tickets without departments
SELECT * FROM tickets
WHERE dept_id IS NULL;

-- Users with Multiple Tickets
SELECT ticket_id,COUNT(*) AS ticket_count
FROM tickets
GROUP BY ticket_id
HAVING COUNT(*) > 1;

def factorial(n):
    fact = 1
    for i in range(1, n+1):
        fact *= i
    print("Factorial:",fact)
